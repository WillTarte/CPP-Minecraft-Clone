81 inline BlockFileData readBlockFile(const std::string &blockFilePath) {
82  std::ifstream blockFile;
83  blockFile.open(blockFilePath);
84 
85  BlockFileData ret;
86 
87  std::string line;
88  while (std::getline(blockFile, line)) {
89 
90  size_t delimPos = line.find(':');
91  std::string identifier = line.substr(0, delimPos);
92  std::string value = line.substr(delimPos + 1, line.size() - delimPos);
93  value.erase(std::remove(value.begin(), value.end(), ' '), value.end());
94 
95  if (identifier.empty() || value.empty() || identifier == "\n" || value == "\n")
96  continue;
97 
98  if (identifier == "ID") {
99  ret.ID = (BlockID) stoi(value);
100  continue;
101  }
102 
103  if (identifier == "name") {
104  ret.blockName = value;
105  }
106 
107  if (identifier == "model") {
108  ret.modelFile = "./resources/models/" + value;
109  continue;
110  }
111 
112  if (identifier == "top" || identifier == "bottom" || identifier == "left" || identifier == "right" ||
113  identifier == "front" || identifier == "back") {
114  ret.cubeFaceFiles.emplace_back("./resources/textures/" + value);
115  ret.textureType = CUBEMAP;
116  continue;
117  }
118  if (identifier == "all") {
119  ret.textureFile = "./resources/textures/" + value;
120  ret.textureType = TEXTURE2D;
121  break;
122  }
123 
124  if (identifier == "breakable") {
125  ret.breakable = value == "true";
126  }
127  }
128 
129  if (ret.ID < 0) {
130  LOG(WARNING) << "Invalid Block ID!";
131  }
132 
133  return ret;
134 }
